153. 寻找旋转排序数组中的最小值

class Solution
{
    /**
     * @param  Integer[]  $nums
     *
     * @return Integer
     */
    function findMin(array $nums): int
    {
        if (empty($nums)) {
            return -1;
        }

        $l = 0;
        $r = count($nums) - 1;

        while ($l < $r) {
            $mid = $l + (($r - $l) >> 1);
            if ($nums[$mid] < $nums[$r]) {
                $r = $mid;
            } else {
                $l = $mid + 1;
            }
        }

        return $nums[$l];
    }
}

162. 寻找峰值

class Solution
{

    /**
     * @param  Integer[]  $nums
     *
     * @return Integer
     */
    function findPeakElement(array $nums): int
    {
        if (empty($nums)) {
            return -1;
        }

        $l = 0;
        $r = count($nums) - 1;

        while ($l < $r) {
            $mid = $l + (($r - $l) >> 1);
            if ($nums[$mid] > $nums[$mid + 1]) {
                $r = $mid;
            } else {
                $l += 1;
            }
        }

        return $l;
    }
}
